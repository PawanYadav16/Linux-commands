# ------------------------------
# Script: Show last 10 failed login attempts

#!/bin/bash

# Step 1: Inform the user
echo "Checking for last 10 failed login attempts..."

# Step 2: Check if log file exists (Ubuntu/Debian uses /var/log/auth.log)
LOG_FILE="/var/log/auth.log"

# Step 3: Verify the file exists
if [ -f "$LOG_FILE" ]; then
                                                  # Step 4: Use grep to find 'Failed password' lines and tail to get last 10
    grep "Failed password" "$LOG_FILE" | tail -n 10
    echo "failed login attempts saved in /var/log/auth.log"
else
    echo "Log file not found! You may not have permission or the file doesn't exist."
fi
